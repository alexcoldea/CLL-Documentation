[docs]def GET_SERVER_IP(TARGET_IP='10.255.255.255', PORT=1):
    """
        returns the server IP
    """
    S = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    try:
        # doesn't even have to be reachable
        S.connect((TARGET_IP, PORT))
        IP = S.getsockname()[0]
    except:
        IP = '127.0.0.1'
    finally:
        S.close()
    return IP
